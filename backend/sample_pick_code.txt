* Simple Pick Basic program to demonstrate migration
* This program reads customer records and prints invoice totals

PROGRAM INVOICE.REPORT

* Variable declarations
DIM CUSTOMER.FILE(100)
TOTAL = 0
COUNTER = 0

* Open customer file
OPEN "CUSTOMERS" TO CUSTOMER.FILE ELSE
  PRINT "Cannot open CUSTOMERS file"
  STOP
END

* Main processing loop
FOR I = 1 TO 100
  READ CUSTOMER.REC FROM CUSTOMER.FILE, I ELSE
    CONTINUE
  END
  
  * Extract invoice amount
  INVOICE.AMT = CUSTOMER.REC<3>
  TOTAL = TOTAL + INVOICE.AMT
  COUNTER = COUNTER + 1
  
  * Print customer info
  PRINT "Customer: ": CUSTOMER.REC<1>
  PRINT "Invoice: $": INVOICE.AMT
NEXT I

* Print summary
PRINT ""
PRINT "Total Records: ": COUNTER
PRINT "Total Amount: $": TOTAL

CLOSE CUSTOMER.FILE
END
